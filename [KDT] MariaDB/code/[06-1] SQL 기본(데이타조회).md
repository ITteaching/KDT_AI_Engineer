### 6.1.1 원하는 데이터를 가져와 주는 기본적인 <SELECT ... FROM>

* SELECT 기본 구문

```SQL
USE employees;

SELECT * FROM titles;

-- 다른 데이타베이스에 있는 테이블 조회
SELECT * FROM employees.titles;

-- 테이블의 특정 컬럼 조회
SELECT first_name, last_name, gender FROM employees;

-- SELECT Alias 예제
SELECT first_name AS fn, last_name LN, gender FROM employees;
```

* database, table 생성

```SQL
-- database 생성
CREATE DATABASE sqlDB;

USE sqlDB;

-- userTbl 테이블 생성
CREATE TABLE userTbl
( userID    CHAR(8) not null primary key, 	-- 사용자 아이디(PK)
  name      VARCHAR(10) NOT NULL, 		-- 이름
  birthYear INT NOT NULL,			-- 출생연도
  addr      CHAR(2) NOT NULL,		-- 지역(서울, 경기, 경남)
  mobile1   CHAR(3),				-- 국번
  mobile2   CHAR(8),				-- 번호2
  height    SMALLINT,			-- 키
  mDate     DATE				-- 회원 가입일
);

--  buyTbl 테이블 생성
CREATE TABLE buyTbl
( num INT AUTO_INCREMENT NOT NULL PRIMARY KEY,	-- 순번(PK)
  userID CHAR(8) NOT NULL, 			 	-- 아이디(FK)
  prodName CHAR(6) NOT NULL,				-- 물품명
  groupName CHAR(4),						-- 분류
  price INT NOT NULL,					-- 단가
  amount SMALLINT NOT NULL,				-- 수량
  FOREIGN KEY (userID) REFERENCES userTbl(userID)			-- 외래 키 지정
);
```

# [실습 3 : 16 ~ 17p]

### 6.1.2 특정 조건 조회

* 조건 연산자 (=, <, >, <=, >=, !=)

```SQL
-- = 연산자
-- 이름이 "김경호"인 사람의 모든 컬럼
SELECT *
  FROM userTbl
 WHERE name = '김경호';

-- >= 연산자
-- 1970년 이후 출생한 사람의 아이디와 이름
SELECT userID, name
  FROM userTbl
 WHERE birthyear >= 1970;

-- != 연산자
-- 주소가 "서울"이 아닌 사람의 이름과 주소
SELECT name, addr
  FROM userTbl
 WHERE addr != '서울';
```

* 관계 연산자 (AND, OR)

```SQL
-- AND 연산자
-- 1970년 이후 출생, 신장 182 이상인 사람의 아이디와 이름
SELECT userID, name
  FROM userTbl
 WHERE birthyear >= 1970
     AND height >= 182;

-- OR 연산자
-- 1970년 이후 출생하거나, 신장 182 이상인 사람의 아이디와 이름
SELECT userID, name
  FROM userTbl
 WHERE birthyear >= 1970
       OR height >= 182;

-- <= AND >=
-- 키가 180에서 183인 사람의 이름과 키
SELECT name, height
  FROM userTbl
 WHERE height >= 180 AND height <= 183;
```



# [실습 4 : 18 ~ 20P]

* 연산자 (BETWEEN, IN)

```SQL
-- BETWEEN [A] AND [B]
-- 키가 180에서 183인 사람의 이름과 키
SELECT name, height
  FROM userTbl
 WHERE height BETWEEN 180 AND 183;

-- IN
-- 지역이 서울, 경기인 사람의 이름과 주소
SELECT name, addr
  FROM userTbl
 WHERE addr in ('서울','경기');

-- IN 연산자를 OR로 변경 시 : IN을 사용하는 것이 정석임
SELECT name, addr
  FROM userTbl
 WHERE addr = '서울' or addr = '경기';
```

* 연산자 (LIKE, %, _)

```SQL
-- LIKE, %
-- 성이 김씨인 사람의 이름과 아이디
SELECT name, userID 
  FROM userTbl
 WHERE name like '김%';

-- LIKE, %
-- 이름에 경자가 포함된 사람의 이름과 아이디
SELECT name, userID 
  FROM userTbl
 WHERE name like '%경%';

-- LIKE, _ (글자 1개 대체 가능)
-- 이름이 종신인 사람의 이름과 아이디
SELECT name, userID 
  FROM userTbl
 WHERE name like '_종신';
```



# [실습 5 : 21P]

* 연산자( ANY / ALL / SOME, SubQuery) 

```SQL
-- SubQuery : 키가 177이상인 사람
  SELECT NAME, height FROM usertbl WHERE height > 177;

-- 이름이 '김경호'인 사람보다 키가 큰 사람
SELECT *
  FROM usertbl
 WHERE height > (SELECT height FROM usertbl WHERE NAME='김경호')

-- 지역이 '경남'인 사람보다 키가 큰 사람 : 에러발생
SELECT *
  FROM usertbl
 WHERE height > (SELECT height FROM usertbl WHERE addr='경남')
-- error message : ERROR 1242 (21000): Subquery returns more than 1 row

-- SOME : 서브쿼리 결과값의 하나의 조건만 만족하면 됨
-- SOME과 ANY는 동일
SELECT *
  FROM usertbl
 WHERE height >= SOME (SELECT height FROM usertbl WHERE addr='경남')

 -- ALL : 서브쿼리 결과값의 모든  조건을 만족해야 함됨
SELECT *
  FROM usertbl
 WHERE height >= ALL (SELECT height FROM usertbl WHERE addr='경남')

 -- "= ANY"와 "IN"은 동일
SELECT *
  FROM usertbl
 WHERE height = ANY (SELECT height FROM usertbl WHERE addr='경남')

SELECT *
  FROM usertbl
 WHERE height IN (SELECT height FROM usertbl WHERE addr='경남')
```

* 소팅(ORDER BY)

```SQL
-- mDate (가입일) : 먼저 가입한 순서로 회원의 이름과 가입일  조회
SELECT NAME, mDate FROM usertbl ORDER BY mDate;

-- mDate (가입일) : 나중에 가입한 순서로 회원의 이름과 가입일  조회
SELECT NAME, mDate FROM usertbl ORDER BY mDate DESC;

-- 신장은 큰 수서로, 같으면 이름순으로 정렬
SELECT NAME, height FROM usertbl ORDER BY height DESC, NAME;
```

* DISTINCT(중복 열 제거), LIMIT(조회 데이타 갯수 제한)

```SQL
-- 거주지역 리스트 출력
SELECT DISTINCT addr FROM usertbl;

-- 조회 건수 5건으로 제한
SELECT * FROM usertbl LIMIT 5;

-- mDate (가입일)이 오래된 직원 5명의 이름과 가입일  출력
SELECT NAME, mDate
  FROM usertbl
ORDER BY mDate
   LIMIT 5;

-- mDate (가입일)이 오래된 직원 중 오래된 순서로 6번째부터 5명의 이름과 가입일  출력   
SELECT NAME, mDate
  FROM usertbl
ORDER BY mDate
   LIMIT 5,5;
```

# [실습 7 : 24 ~ 27p]

### 6.1.3 GROUP BY 및 HAVING 그리고 집계 함수

* 집계 SELECT문 기본 형식 

```SQL
SELECT 컬럼, 컬럼, 집계 함수
  FROM 테이블명
 WHERE 조건절
 GROUP BY 컬럼, 컬럼 ...
 HAVING 집계 함수 조건
 ORDER BY 컬럼, 컬럼 ... [ASC | DESC]
```

* 집계 함수 예제 : userid별 구매액 집계

```SQL
-- 구매 테이블 확인: userid 순으로
SELECT *
  FROM buytbl
 ORDER BY userid;

-- userid별로 구매 갯수 구하기
SELECT userid, SUM(amount)
  FROM buytbl 
 GROUP BY userid;

-- 별칭 사용하여 결과 보기
SELECT userid AS '사용자 아이디', SUM(amount) AS '총 구매 갯수'
  FROM buytbl
GROUP BY userid;

-- userid별 총 구매 갯수, 총 금액 조회하기
SELECT userid AS '사용자 아이디', SUM(amount) AS '총 구매 갯수',SUM(amount * price) AS '총 구매액'
  FROM buytbl
GROUP BY userid;
```

* 집계 함수

  : AVG(), MIN(), MAX(), COUNT(), COUNT(DISTINCT), STDEV(), VAR_SAMP()

```SQL
-- AVG() : 평균값
-- 전체 평균 구하기 : buytbl
SELECT AVG(amount) AS '평균 구매 개수' 
  FROM buytbl;

-- userID별 평균 구하기 : buytbl
SELECT userID, AVG(amount) 
  FROM buytbl
GROUP BY userID;

-- MAX() : 최대값, MIN() : 최소값
-- 가장 키가 큰 회원과 가장 작은 회원의 키
SELECT MAX(height), MIN(height)
  FROM usertbl 

-- 가장 키가 크거나 작은 회원의 이름과 키 출력
SELECT NAME, height
  FROM usertbl
 WHERE height = (SELECT MAX(height) FROM usertbl)
    OR height = (SELECT MIN(height) FROM usertbl)
    
-- 휴대폰번호가 있는 사용자의 수    
SELECT COUNT(*)
  FROM usertbl
 WHERE mobile1 IS NOT NULL

SELECT COUNT(mobile1)
  FROM usertbl
```

* userid별 총 구매 갯수, 총 금액 조회하기

```SQL
-- 총 구매액이 1000 이상인 사용자만 조회
-- 총 구매액이 큰 순서부터 조회
SELECT userid AS '사용자 아이디', SUM(amount) AS '총 구매 갯수',SUM(amount * price) AS '총 구매액'
  FROM buytbl
GROUP BY userid
HAVING SUM(amount * price) > 1000
ORDER BY SUM(amount * price) DESC

-- [참고만] 총합, 중간합계 : WITH ROLLUP
-- GROUP BY 절 뒤에 WITH ROLLUP 구문 사용
-- 중간합계 있는 경우
SELECT num, groupname, SUM(price * amount)
  FROM buytbl
GROUP BY groupname, num
WITH ROLLUP

-- [참고만] 총합만 구하는 경우
SELECT groupname, SUM(price * amount)
  FROM buytbl
GROUP BY groupname
WITH ROLLUP
```
